====================== User algo upon receiving a message: ======================

Message_t WANT_TO_CONNECT
Message_t CONNECT_TO
Message_t ANTENNA_REPLY_CONNECT
Message_t CHECK_CONNEXION
Message_t CONNEXION_OK
Message_t MESSAGE

boolean trying_to_connect = false
boolean connected = false
timestamp last_ack_time

// Rule 1 : We have a new user, so this user want to connect to an antenna
Manual_starting* ? && !trying_to_connect => out_broadcast_user_antennas ! WANT_TO_CONNECT(id, coord);
                                          trying_to_connect = true;

// Rule 2 : We get a reply from an antenna that want to connect to this user
in ? ANTENNA_REPLY_CONNECT(id, coord) => antenna_replies = antenna_replies + [in];

// Rule 3 : After a collection delay (500ms), we connect to the closest one
timer_finished** => connected_to = min(antenna_replies);
                    out_broadcast_user_antennas ! CONNECT_TO(id, connected_to);
                    trying_to_connect = false;
                    connected = true;
                    last_ack_time = now;

// Rule 4 : This user want to send a message to another user
Send_msg(content, to)*** ? && connected => connected_to ! MESSAGE(id, to, content, coord);

// Rule 5 : This user want to send a message, but is currently not connected
Send_msg(content, to)*** ? && !connected && !trying_to_connect => Rule 1;

// Rule 6 : User moves position
Manual_move(new_coord) => coord = new_coord;

// Rule 7 : Heartbeat timer (every 5s)
delay_message >= 5s && connected => connected_to ! CHECK_CONNEXION(id, coord);

// Rule 8 : Receiving acknowledgement from antenna
in ? CONNEXION_OK => last_ack_time = now;
                    connected = true;

// Rule 9 : Connection lost detection (timeout 12s)
now - last_ack_time > 12s && connected => connected = false;
                                         Rule 1 (reconnect);

// Rule 10 : Receiving a message for us
in ? MESSAGE(sender, to, content) && to == id => display(content);

====================== Antenna algo upon receiving a message: ======================

// Rule 1 : Connection request from user (closed enough)
in_broadcast ? WANT_TO_CONNECT(id, coord_user) && distance(coord_user, coord) < threshold 
           => user(id) ! ANTENNA_REPLY_CONNECT(id, coord);

// Rule 2 : A user confirms connection to us
in_broadcast ? CONNECT_TO(user_id, target_antenna) && target_antenna == id 
           => user_list = user_list + user_id;

// Rule 3 : A user confirms connection to ANOTHER antenna
in_broadcast ? CONNECT_TO(user_id, target_antenna) && target_antenna != id 
           => user_list = user_list - user_id;

// Rule 4 : User heartbeat (closed enough)
in ? CHECK_CONNEXION(user_id, coord_user) && distance(coord_user, coord) < threshold
           => user(user_id) ! CONNEXION_OK;

// Rule 5 : User heartbeat (too far)
in ? CHECK_CONNEXION(user_id, coord_user) && distance(coord_user, coord) >= threshold
           => user_list = user_list - user_id;

// Rule 6 : Receive MESSAGE (Receiver is local)
in ? MESSAGE(sender, receiver, content) && receiver in user_list 
           => user(receiver) ! MESSAGE(sender, receiver, content);

// Rule 7 : Receive MESSAGE (Receiver not local, not original antenna)
in ? MESSAGE(sender, receiver, content) && receiver not in user_list && original_antenna != id
           => next_antenna_left ! MESSAGE(sender, receiver, content, original_antenna = (id if null));

// Rule 8 : Receive MESSAGE (Receiver not found, back at original antenna)
in ? MESSAGE(sender, receiver, content) && receiver not in user_list && original_antenna == id
           => master_node ! MESSAGE(sender, receiver, content);

====================== MasterAntenna algo upon receiving a message: ======================

// Rule 1 : Store undelivered message
in ? MESSAGE(sender, receiver, content) => message_storage(receiver) = message_storage(receiver) + [message];

// Rule 2 : User reconnected somewhere in the ring
in_broadcast ? CONNECT_TO(user_id, antenna_id) && user_id in message_storage
           => antenna(antenna_id) ! stored_messages(user_id);
              message_storage(user_id) = empty;
