in ? ANTENNA_REPLY_CONNEXT && num_antenna_replies == 2 => connected_to = min**(antenna_replies);
                                                          out_broadcast_user_antennas ! CONNECT_TO(id, connected_to);

// Rule 4 : This user want to send a message to another user, and is currently connected to an antenna, so this user send
// the message to that antenna
Send_msg(content, to)*** ? && connected => connected_to ? MESSAGE(content, to);

// Rule 5 : This user want to send a message to another user, but is currently not connected to an antenna, so the message is

// the message is not sent
Send_msg(content, to)*** ? && !connected && trying_to_connect => ;
